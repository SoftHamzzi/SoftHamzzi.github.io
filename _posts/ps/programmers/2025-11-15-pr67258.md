    int minSize = INT_MAX;
    
    int l=0, r=0;

            minSize = r-l+1;
            minRange = {l+1, r+1};
        }

            string temp = gems[l++];
            um[temp]--;
            if(um[temp] == 0) um.erase(temp);
        } else {
            um[gems[++r]]++;
        }
    }
    
    return minRange;
}

vector<int> solution(vector<string> gems) {
    int cnt = getCnt(gems);
    return getShortestRange(gems, cnt);
